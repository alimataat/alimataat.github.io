创建所需目录与用户

  1. groupadd mysql
  2. useradd -g mysql mysql -s /sbin/nologin
  3. mkdir -p /service/logs/software/mysql_3306/
  4. mkdir -p /service/databases/mysql_3306/data/
  5. chown -R mysql. /service/logs/software/mysql_3306/
  6. chown -R mysql. /service/databases/mysql_3306/data/

  10. get_read_only=`$comm -e "show variables like 'read_only'\G"| grep -w "Value" |awk '{print $NF}'`

  12. if [ $slave_info -ne 0 ]
  13. then
  14.    echo "MySQL $port  Instance is Slave........"
  15.    $comm -e "show slave status\G" | egrep -w "Master_Host|Master_User|Master_Port|Master_Log_File|Read_Master_Log_Pos|Relay_Log_File|Relay_Log_Pos|Relay_Master_Log_File|Slave_IO_Running|Slave_SQL_Running|Exec_Master_Log_Pos|Relay_Log_Space|Seconds_Behind_Master"
  16.    exit 2
  17. fi

  19. if [ -z $value ]
  20. then
  21.    exit 2
  22. fi

  24. if  [ $get_read_only = "ON" ]
  25. then
  26.     echo "MySQL $port  Instance is read only........"
  27.     exit 2
  28. fi

  30. if  [ -n$value -a  $get_read_only = "OFF" ]
  31. then
  32.     echo "MySQL $port  Instance is Master........"
  33.     exit 0
  34. fi

  36. exit 2
